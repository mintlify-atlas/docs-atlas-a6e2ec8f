## Configuration

### OpenAI setup

Add your OpenAI API key to `.env.local`:

```bash .env.local
OPENAI_API_KEY=sk-...
```

<Warning>
Never commit your OpenAI API key to version control. Use environment variables only.
</Warning>

### OpenAI client

**Location:** `lib/openai.ts`

```typescript lib/openai.ts
import OpenAI from 'openai';

let openaiClient: OpenAI | null = null;

export function getOpenAIClient(): OpenAI {
  if (!openaiClient) {
    const apiKey = process.env.OPENAI_API_KEY;
    
    if (!apiKey) {
      throw new Error('OPENAI_API_KEY is not configured');
    }

    openaiClient = new OpenAI({ apiKey });
  }

  return openaiClient;
}

export function isOpenAIConfigured(): boolean {
  return !!process.env.OPENAI_API_KEY;
}
```

## Generating insights

### API endpoint

**Location:** `app/api/ai/manager-report/route.ts`

```typescript
export async function POST(request: NextRequest) {
  // 1. Authenticate user
  const session = await getSession();
  if (!session) {
    return NextResponse.json(
      { error: 'Unauthorized', fallback: true },
      { status: 401 }
    );
  }

  // 2. Check if OpenAI is configured
  if (!isOpenAIConfigured()) {
    return NextResponse.json(
      { 
        error: 'AI features are not available',
        suggestion: 'Please add OPENAI_API_KEY to your .env.local file.'
      },
      { status: 503 }
    );
  }

  // 3. Parse request data
  const data: AdvancedAnalysisResponse = await request.json();

  // 4. Check cache
  const cacheKey = generateCacheKey(data);
  const cached = cache.get(cacheKey);
  
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return NextResponse.json({ 
      insights: cached.data,
      cached: true
    });
  }

  // 5. Generate AI insights
  const insights = await generateManagerInsights(data);

  // 6. Cache the result
  cache.set(cacheKey, { data: insights, timestamp: Date.now() });

  return NextResponse.json({ insights, cached: false });
}
```

### OpenAI API call

```typescript
async function generateManagerInsights(data: AdvancedAnalysisResponse) {
  const openai = getOpenAIClient();
  const prompt = buildPrompt(data);

  const completion = await openai.chat.completions.create({
    model: 'gpt-5-mini',
    messages: [
      {
        role: 'system',
        content: `You are an expert engineering manager analyzing team productivity.
        Provide actionable insights in a professional, concise manner.
        Always respond with valid JSON only, no markdown formatting.`
      },
      {
        role: 'user',
        content: prompt
      }
    ],
    temperature: 1,
    max_completion_tokens: 2000,
    response_format: { type: "json_object" }
  });

  const content = completion.choices[0]?.message?.content;
  return parseAIResponse(content);
}
```

<Info>
The AI uses `response_format: { type: "json_object" }` to ensure structured JSON output.
</Info>

## Caching system

AI insights are cached to reduce API costs and improve performance.

### Cache implementation

```typescript
const cache = new Map<string, { data: any; timestamp: number }>();
const CACHE_TTL = 1000 * 60 * 60; // 1 hour

function generateCacheKey(data: AdvancedAnalysisResponse): string {
  const { timeline, metadata } = data;
  return `ai_insights_${timeline.totalCommits}_${metadata.dateRange.start}_${metadata.dateRange.end}`;
}
```

<Tabs>
  <Tab title="Cache hit">
    If cached insights exist and are less than 1 hour old:
    
    ```typescript
    if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
      return NextResponse.json({ 
        insights: cached.data,
        cached: true  // Indicates cache hit
      });
    }
    ```
  </Tab>
  
  <Tab title="Cache miss">
    If no cache or expired, generate new insights:
    
    ```typescript
    const insights = await generateManagerInsights(data);
    cache.set(cacheKey, { data: insights, timestamp: Date.now() });
    
    return NextResponse.json({ 
      insights,
      cached: false
    });
    ```
  </Tab>
</Tabs>

<Note>
In production, consider using Redis or another persistent cache instead of in-memory Map.
</Note>
